37 void GUIAnalyzeOutputWindow::Update() {
38 
39  //zurücksetzen der Tabelle
40  table->ClearGrid();
41  table->DeleteRows(0, table->GetRows(), true);
42  table->DeleteCols(0, table->GetCols(), true);
43  table->CreateGrid(0, 0);
44 
45  //Erstellen der Zeilenbezeichnungen
46  table->SetRowLabelValue(0, wxT("Objekt:"));
47  table->SetRowLabelValue(1, wxT("Sensordatensatz:"));
48  table->SetRowLabelValue(2, wxT("Volumen in m³:"));
49  table->SetRowLabelValue(3, wxT("Energie in kJ:"));
50  table->SetRowLabelValue(4, wxT("Material:"));
51  table->SetRowLabelValue(5, wxT("Volumen in m³:"));
52  table->SetRowLabelValue(6, wxT("Energie in kJ:"));
53 
54  //Setzen der Tabelleneinstellungen
55  table->SetRowLabelSize(150);
56  table->SetColLabelSize(0);
57  table->SetDefaultColSize(100, true);
58 
59  //Anzahl der dargestellten Objekt- und Materialeigenschaften
60  const int OBJECT_PROPERTY_COUNT = 3;
61  const int MATERIAL_PROPERTY_COUNT = 3;
62 
63  //erstellen der Zeilen der Tabelle
64  table->AppendRows(OBJECT_PROPERTY_COUNT + MATERIAL_PROPERTY_COUNT + 1, true);
65 
66  //Breite der Tabelle in Zellen (für Zellenindices benötigt)
67  int all_mat_cell_count = 0;
68 
69  //Für alle Objekte...
70  for (unsigned int o = 0; o < wxGetApp().getDataObjects()->size(); o++) {
71  //aktuelles Objekt
72  ObjectData* obj = wxGetApp().getDataObjects()->at(o);
73  //Objekt zum speichern der Analysedaten
74  Analyzer::AnalyzerData_object data;
75 
76  //Analysieren des Objekts
77  Analyzer analyzer;
78  analyzer.analyzeObject(wxGetApp().getDataObjects()->at(o), &data);
79 
80  //Anzahl der Materialien des Objekts
81  int matcount = obj->getMaterials()->size();
82  //Anzahl der Sensordatensätze des Objekts
83  int sdcount = data.data_sets.size();
84 
85  //Die benötigten Spalten hinzufügen
86  table->AppendCols(sdcount * matcount, true);
87 
88  //Die Spalte für den Objektnamen auf volle Breite vergrößern
89  table->SetCellSize(0, all_mat_cell_count, 1, matcount * sdcount);
90  //Objektnamen eintragen
91  table->SetCellValue(0, all_mat_cell_count,
92  wxString::FromAscii(obj->getName().c_str()));
93  // Objektnamen zentrieren
94  table->SetCellAlignment(wxALIGN_CENTRE, 0, all_mat_cell_count);
95 
96  //Anzahl der bereits mit Materialdaten gefüllten Zellen
97  int materialcells = 0;
98 
99  //Für alle Sensordatensätze
100  for (int s = 0; s < sdcount; s++) {
101  //Analysedaten für einen Datensatz
102  Analyzer::AnalyzerData_dataset* data_set = &data.data_sets.at(s);
103 
104  //Zellen für den Datensatznamen über alle Materialzellen vergrößern
105  for (int i = 0; i < OBJECT_PROPERTY_COUNT; i++) {
106  table->SetCellSize(i + 1, all_mat_cell_count + s * matcount, 1,
107  matcount);
108  };
109 
110  //Den Namen des Datensatzes eintragen
111  table->SetCellValue(1, all_mat_cell_count + s * matcount,
112  wxString::FromAscii(data_set->name.c_str()));
113  //Den Namen des Datensatzes zentrieren
114  table->SetCellAlignment(wxALIGN_CENTRE, 1,
115  all_mat_cell_count + s * matcount);
116 
117  //Die Analyseergebnisse für den Sensordatensatz eintragen
118  table->SetCellValue(2, all_mat_cell_count + s * matcount,
119  wxString::FromAscii(floattostr(data.volume).c_str()));
120  table->SetCellValue(3, all_mat_cell_count + s * matcount,
121  wxString::FromAscii(floattostr(data_set->heat_energy).c_str()));
122 
123  //Für alle Materialien...
124  for (int i = 0; i < matcount; i++) {
125  //Analysedaten für einen Material
126  Analyzer::AnalyzerData_material* mat = &data_set->mat_data.at(i);
127  //Zelle Mitzählen
128  materialcells++;
129  //Materialnamen eintragen
130  table->SetCellValue(OBJECT_PROPERTY_COUNT + 1,
131  all_mat_cell_count + s * matcount + i,
132  wxString::FromAscii(mat->name.c_str()));
133  //Materialnamen zentrieren
134  table->SetCellAlignment(wxALIGN_CENTRE, OBJECT_PROPERTY_COUNT + 1,
135  all_mat_cell_count + s * matcount + i);
136 
137  //Die Analyseergebnisse für das Material eintragen
138  table->SetCellValue(OBJECT_PROPERTY_COUNT + 2,
139  all_mat_cell_count + s * matcount + i,
140  wxString::FromAscii(floattostr(mat->volume).c_str()));
141  table->SetCellValue(OBJECT_PROPERTY_COUNT + 3,
142  all_mat_cell_count + s * matcount + i,
143  wxString::FromAscii(floattostr(mat->heat_energy).c_str()));
144  }
145  }
146  //Materialzellen mitzählen
147  all_mat_cell_count += materialcells;
148  }
149 }

151 void GUIAnalyzeOutputWindow::ToClipboard() {
152 
153  wxString copy_data;
154  bool something_in_this_line;
155 
156  copy_data.Clear();
157 
158  //Kopieren der Tabelleninhalte
159  for (int i=0; i<table->GetRows();i++) {
160 
161  something_in_this_line = false;
162  for (int j=0; j<table->GetCols(); j++) {
163 
164  if (table->IsInSelection(i,j)) {
165 
166  if (something_in_this_line == false) {
167 
168  if (copy_data.IsEmpty() == false) {
169  copy_data.Append(wxT("\r\n")); // in windows if copy to notepad need \r\n to newline
170  }
171 
172  something_in_this_line = true;
173  } else {
174  copy_data.Append(wxT("\t"));
175  }
176 
177  copy_data = copy_data + table->GetCellValue(i,j);
178  }
179  }
180  }
181 
182  //Schreiben in die Zwischenablage
183  if (wxTheClipboard->Open()) {
184  wxTheClipboard->SetData(new wxTextDataObject(copy_data));
185  wxTheClipboard->Close();
186  }
