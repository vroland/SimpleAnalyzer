30 void GUIAnalyzeOutputWindow::Update() {
31  table->ClearGrid();
32  table->DeleteRows(0,table->GetRows(), true);
33  table->DeleteCols(0,table->GetCols(), true);
34  table->CreateGrid(0,0);
35  table->SetRowLabelValue(0,wxT("Objekt:"));
36  table->SetRowLabelValue(1,wxT("Sensordatensatz:"));
37  table->SetRowLabelValue(2,wxT("Volumen in m³:"));
38  table->SetRowLabelValue(3,wxT("Energie in kJ:"));
39  table->SetRowLabelValue(4,wxT("Material:"));
40  table->SetRowLabelValue(5,wxT("Volumen in m³:"));
41  table->SetRowLabelValue(6,wxT("Energie in kJ:"));
42  table->SetRowLabelSize(150);
43  table->SetColLabelSize(0);
44  table->SetDefaultColSize(100,true);
45  int obj_prop_count = 3;
46  int material_prop_cout = 3;
47  int all_sd_count = 0;
48  int all_mat_cell_count = 0;
49  table->AppendRows(obj_prop_count+material_prop_cout+1,true);
50  for (unsigned int o=0;o<data_objects.size();o++) {
51  ObjectData* obj = data_objects.at(o);
52  Analyzer::AnalyzerData_object data;
53  Analyzer analyzer;
54  analyzer.analyzeObject(data_objects.at(o),&data);
55  int matcount = obj->materials.size();
56  int sdcount = data.data_sets.size();
57  table->AppendCols(sdcount*matcount,true);
58  table->SetCellSize(0,all_mat_cell_count , 1, matcount*sdcount);
59  table->SetCellValue(0,all_mat_cell_count,wxString::FromAscii(obj->name.c_str()));
60  table->SetCellAlignment(wxALIGN_CENTRE,0,all_mat_cell_count);
61  int materialcells = 0;
62  for (int s=0;s<sdcount;s++) {
63  Analyzer::AnalyzerData_dataset* data_set = &data.data_sets.at(s);
64  for (int i=0;i<obj_prop_count;i++) {
65  table->SetCellSize(i+1,all_mat_cell_count+s*matcount , 1, matcount);
66  };
67  table->SetCellValue(1,all_mat_cell_count+s*matcount,wxString::FromAscii(data_set->name.c_str()));
68  table->SetCellAlignment(wxALIGN_CENTRE,1,all_mat_cell_count+s*matcount);
69  table->SetCellValue(2,all_mat_cell_count+s*matcount,wxString::FromAscii(ftostr(data.volume).c_str()));
70  table->SetCellValue(3,all_mat_cell_count+s*matcount,wxString::FromAscii(ftostr(data_set->heat_energy).c_str()));
71 
72  for (int i=0;i<matcount;i++) {
73  Analyzer::AnalyzerData_material* mat = &data_set->mat_data.at(i);
74  materialcells++;
75  table->SetCellValue(obj_prop_count+1,all_mat_cell_count+s*matcount+i,wxString::FromAscii(mat->name.c_str()));
76  table->SetCellAlignment(wxALIGN_CENTRE,obj_prop_count+1,all_mat_cell_count+s*matcount+i);
77  table->SetCellValue(obj_prop_count+2,all_mat_cell_count+s*matcount+i,wxString::FromAscii(ftostr(mat->volume).c_str()));
78  table->SetCellValue(obj_prop_count+3,all_mat_cell_count+s*matcount+i,wxString::FromAscii(ftostr(mat->heat_energy).c_str()));
79  }
80  }
81  all_mat_cell_count+=materialcells;
82  all_sd_count += sdcount;
83  }
84 }
