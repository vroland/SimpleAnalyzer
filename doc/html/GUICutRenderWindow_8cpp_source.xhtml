166 void render_thread(bool* status_flag, float* value_img, wxImage* image,
167  int width, int height, int startheight, int delta_h,
168  CutRender_info* info, Vector3D* xvec, Vector3D* yvec, Vector3D* v0,
169  vector<tetgenio*>* bases, ObjectData* obj,
170  vector<SensorPoint>* sensor_data, bool use_last_tet) {
171 
172  Interpolator interpolator;
173  //Referenz auf den letzten für die Interpolation gewählten Tetraeder
174  vector<SensorPoint*>* last_tet = new vector<SensorPoint*>;
175  //Referenz auf den aktuell für die Interpolation gewählten Tetraeder
176  vector<SensorPoint*>* new_tet = new vector<SensorPoint*>;
177  //Die Visualisierungsinformationen
178  Visualization_info* vis_info = wxGetApp().getVisualizationInfo();
179 
180  //Wurde der zuletzt verwendete Tetraeder initialisiert?
181  bool last_tet_init = false;
182  //Für alle Pixel im zu berechnenden Streifen...
183  for (int x = 0; x < width; x++) {
184  for (int y = startheight; y < startheight + delta_h; y++) {
185 
186  //Position des Pixels im 3D-Raum
187  Vector3D* p = v0->copy();
188 
189  //Verschiebung auf der X-Achse der Ebene im vergleich zum Ebenenmittelpunkt
190  Vector3D* part_x = xvec->copy();
191  part_x->mult(
192  x * info->mmperpixel / 1000.
193  - width * info->mmperpixel / 2000);
194 
195  //Verschiebung auf der Y-Achse der Ebene im vergleich zum Ebenenmittelpunkt
196  Vector3D* part_y = yvec->copy();
197  part_y->mult(
198  y * info->mmperpixel / 1000.
199  - height * info->mmperpixel / 2000);
200 
201  //Berechnen der Position des Pixels im 3D-Raum
202  p->add(part_x);
203  p->add(part_y);
204  delete part_x;
205  delete part_y;
206 
207  //Initialisieren des Pixels
208  image->SetAlpha(x, y, 0);
209  image->SetRGB(x, y, 0, 0, 0);
210  value_img[y * width + x] = -300;
211 
212  //Für alle Materialien des objekts...
213  for (unsigned int m = 0; m < obj->getMaterials()->size(); m++) {
214  //Das aktuelle Material
215  ObjectData::MaterialData* mat = &obj->getMaterials()->at(m);
216 
217  //Befindet sich der Punkt im aktuellen Material?
218  if (pointInsideMesh(p, bases->at(m), info->in_volume_algorithm)) {
219 
220  //Ermitteln des Wertes für den Punkt
221  int status = 0;
222  interpolator.setMode(mat->interpolation_mode);
223  float value = (float) getPointValue(status, sensor_data,
224  p->getXYZ(), &interpolator,
225  (last_tet_init && use_last_tet) ? last_tet : NULL,
226  use_last_tet ? new_tet : 0);
227 
228  //Tausch der Speicherorte des Tetraeders, in dem sich der Punkt in diesem und im vorherigen Durchlauf befand
229  vector<SensorPoint*>* temp;
230  temp = last_tet;
231  last_tet = new_tet;
232  new_tet = temp;
233  last_tet_init = true;
234 
235  //Speichern des Wertes in der Temperaturverteilung
236  value_img[y * width + x] = value;
237 
238  //Berechnen der Farbe zum Temperaturwert
239  float inverse_hue = (value - vis_info->min_visualisation_temp)
240  / (vis_info->max_visualisation_temp
241  + vis_info->min_visualisation_temp);
242  float* color = hsvToRgb((1.0 - clampHue(inverse_hue)) * .666, 1, 1);
243 
244  //Speichern des Wertes in der Grafik
245  image->SetRGB(x, y, (unsigned char) (color[0] * 255),
246  (unsigned char) (color[1] * 255),
247  (unsigned char) (color[2] * 255));
248  delete color;
249  image->SetAlpha(x, y, 255);
250  break;
251  }
252  }
253  delete p;
254  }
255  }
256  delete new_tet;
257  delete last_tet;
258 
259  //markieren des Threads als beendet
260  *status_flag = 0;
261 }

263 void GUICutRenderWindow::renderImage(wxImage* image) {
264 
265  //Speichern der Zeit für die Laufzeitmessung
266  timeval tm1;
267  gettimeofday(&tm1, NULL);
268 
269  int width = imgWidthEdit->GetValue();
270  int height = imgHeightEdit->GetValue();
271 
272  //aktualisieren der Skala
273  canvas->getScalePanel()->refresh(width, height);
274 
275  //die Anzahl der zur Berechnung zu verwendenden Kerne
276  core_count = threadcountedit->GetValue();
277 
278  /*
279  * Versuchen, die Interpolation durch vorgezogenes Testen des zuletzt verwendeten Tetraeders zu beschleunigen.
280  * Diese Option ist verursacht Ungenauigkeiten und bietet zumeist wenig Performancegewinn.
281  * Sie ist deshalb standardmäßig deaktiviert und nicht über die Programmoberfläche aktivierbar.
282  */
283  bool use_last_tet = false;
284 
285  //zurücksetzen der Grafik
286  delete image;
287  image = new wxImage(width, height, true);
288  image->InitAlpha();
289 
290  //Punkt als Dezimaltrennzeichen
291  setlocale(LC_NUMERIC, "C");
292 
293  //Eigenschaften der Temperaturverteilung
294  CutRender_info* info = getCutRenderProperties();
295  Triangle* tri = info->tri;
296  //X-Achse der Ebene im 3D-Raum
297  Vector3D* xvec = tri->getV2()->copy();
298  xvec->sub(tri->getV1());
299  //Normale der Ebene
300  Vector3D* tri_nor = tri->getNormal();
301  //Y-Achse der Ebene im 3D-Raum
302  Vector3D* yvec = xvec->crossProduct(tri_nor);
303  delete tri_nor;
304  xvec->normalize();
305  yvec->normalize();
306 
307  //Das aktive Objekt
308  ObjectData* obj = wxGetApp().getActiveObject();
309 
310  //Erstellen möglichst einfacher Geometrien für die Materialien
311  vector<tetgenio*> bases(obj->getMaterials()->size());
312  for (unsigned int i = 0; i < obj->getMaterials()->size(); i++) {
313  tetgenio* tri_io = new tetgenio();
314  string args = "Q";
315  tetrahedralize(const_cast<char*>(args.c_str()),
316  obj->getMaterials()->at(i).tetgeninput, tri_io, NULL, NULL);
317  bases.at(i) = tri_io;
318  }
319 
320  //Zurücksetzen des Datenarrays für die Temperaturverteilung
321  if (value_img != NULL) {
322  delete[] value_img;
323  }
324  value_img = new float[width * height];
325 
326  //Verknüpfen der Ausgabe mit den Temperaturverteilungsdaten und der Grafik
327  canvas->setImage(image);
328  canvas->setValueImg(value_img);
329 
330  //Erstellen des Statusregisters für die Berechnungsthreads
331  bool thread_running[core_count];
332  for (int i = 0; i < core_count; i++) {
333  thread_running[i] = 1;
334  }
335 
336  //Array für die Thread-Objekts
337  vector<thread*> threads = vector<thread*>(0);
338  //Array für die Kopierten Sensordaten. Das Kopieren ist erforderlich, da die Threads die Daten verändern (sortieren).
339  vector<vector<SensorPoint>> copied_sensor_data =
340  vector<vector<SensorPoint>>(core_count);
341 
342  //Höhe für die Streifen, die die einzelnen Threads berechnen
343  int delta_h = height / core_count;
344 
345  //Für alle Threads...
346  for (int i = 0; i < core_count; i++) {
347  //Die Starthöhe des Threads in der Temperaturverteilung
348  int startheight = delta_h * i;
349 
350  //eventuelle Korrektur der Streifenhöhe für den letzten Thread
351  if (i == core_count - 1) {
352  if (startheight + delta_h < height) {
353  delta_h = height - startheight;
354  }
355  }
356 
357  //Aktueller Sensordatensatz
358  SensorData* dataset = &obj->getSensorDataList()->at(
359  obj->getCurrentSensorIndex());
360  vector<SensorPoint>* original_sd = &dataset->data.at(
361  dataset->current_time_index);
362  copied_sensor_data.at(i).resize(original_sd->size());
363 
364  //Kopieren der Sensordaten den Thread
365  for (int p = 0; p < int(original_sd->size()); p++) {
366  copySensorPoint(&original_sd->at(p),
367  &copied_sensor_data.at(i).at(p));
368  }
369 
370  //Starten des Threads
371  threads.resize(threads.size() + 1,
372  new thread(render_thread, &thread_running[i], value_img, image,
373  width, height, startheight, delta_h, info, xvec, yvec,
374  tri->getV1(), &bases, obj, &copied_sensor_data.at(i),
375  use_last_tet));
376  }
377 
378  //Pfüfzahl, ob noch Threads laufen
379  unsigned int running = 0;
380 
381  //Solange threads laufen...
382  do {
383  //Ausgabe aktualisieren
384  canvas->Update();
385  canvas->Refresh();
386 
387  //ermitteln der Pfüfzahl, ob noch Threads laufen. Wenn diese 0 bleibt, sind alle Threads fertig.
388  running = 0;
389  for (int i = 0; i < core_count; i++) {
390  running = running << 1;
391  running += thread_running[i];
392  };
393  } while (running);
394 
395  //Threads zusammenführen
396  for (int i = 0; i < core_count; i++) {
397  threads.at(i)->join();
398  delete threads.at(i);
399  }
400 
401  //Freigeben des Ebenendreiecks
402  for (int i = 0; i < 3; i++) {
403  delete tri->getVert(i);
404  }
405  delete tri;
406 
407  //Zeitmessung beenden
408  timeval tm2;
409  gettimeofday(&tm2, NULL);
410 
411  unsigned long long t = 1000 * (tm2.tv_sec - tm1.tv_sec)
412  + (tm2.tv_usec - tm1.tv_usec) / 1000;
413 
414  //Ausgeben der Berechnungsdauer auf
415  SetTitle(wxT("Berechnung abgeschlossen. ( ") + floattowxstr(t / 1000.)+wxT( "s )"));
416 
417  //Freigeben der Ressourcen
418  delete xvec;
419  delete yvec;
420  for (unsigned int i = 0; i < obj->getMaterials()->size(); i++) {
421  delete bases.at(i);
422  }
423  delete info;
424 }

505  _(""), _(""), _("Portable Network Graphics (*.png)|*.png"),
506  wxFD_SAVE | wxFD_OVERWRITE_PROMPT);
507 
508  //Speichern bestätigt?
509  if (SaveDialog->ShowModal() == wxID_OK) {
510  //Skalenbild kopieren
511  wxImage* scale_img = canvas->getScalePanel()->getImage();
512  //Temperaturverteilungsgrafik kopieren
513  wxImage cut_img = image->Copy();
514 
515  //Für alle Pixel der Temperaturverteilung
516  for (int x = 0; x < cut_img.GetWidth(); x++) {
517  for (int y = 0; y < cut_img.GetWidth(); y++) {
518  //Transparenz des Skalenbildes
519  float alpha = scale_img->GetAlpha(x, y) / 255.;
520 
521  //Kombinieren der Farbwerte
522  cut_img.SetRGB(x, y,
523  cut_img.GetRed(x, y) * (1 - alpha)
524  + scale_img->GetRed(x, y) * alpha,
525  cut_img.GetGreen(x, y) * (1 - alpha)
526  + scale_img->GetGreen(x, y) * alpha,
527  cut_img.GetBlue(x, y) * (1 - alpha)
528  + scale_img->GetBlue(x, y) * alpha);
529 
530  //Kombinieren der Transparenzinformation
531  int new_alpha = cut_img.GetAlpha(x, y) + alpha * 255;
532  new_alpha = new_alpha > 255 ? 255 : new_alpha;
533  cut_img.SetAlpha(x, y, new_alpha);
534  }
535  }
536 
537  //exportieren des Bildes
538  cut_img.SaveFile(SaveDialog->GetPath(), wxBITMAP_TYPE_PNG);
539  }
540 
541  SaveDialog->Close();
542  SaveDialog->Destroy();
