112 void GUITimeline::findMaxValue(ObjectData* obj, bool fast) {
113 
114  //aktueller Wert
115  int val = getValue();
116  //Bereich, in dem Gesucht werden soll
117  int marker1 = getMinValue() - 1;
118  int marker2 = getMaxValue() + 1;
119 
120  //für alle markierten Stellen...
121  for (size_t i = 0; i < markers->size(); i++) {
122  int marker = markers->at(i);
123 
124  //ist der aktuelle Zeitpunkt markiert?
125  if (val == marker) {
126  wxMessageBox(
127  wxT("Der Cursor muss sich zwischen zwei Markierungen befinden!"),
128  wxT("Fehler"));
129  return;
130  }
131 
132  //finden der eingrenzenden markierten Stellen
133  if (marker < val && marker > marker1) {
134  marker1 = marker;
135  }
136 
137  if (marker > val && marker < marker2) {
138  marker2 = marker;
139  }
140  }
141 
142  //eingrenzende markierte Stellen im gültigen Bereich?
143  if (marker1 == getMinValue() - 1 || marker2 == getMaxValue() + 1) {
144  wxMessageBox(
145  wxT("Der Cursor muss sich zwischen zwei Markierungen befinden!"),
146  wxT("Fehler"));
147  return;
148  }
149 
150  //aktuelles Energiemaximum
151  double max_energy = -1;
152  //aktueller Index des Energiemaximums
153  int max_index = marker1;
154 
155  //schnelle Methode verwenden?
156  if (fast) {
157 
158  //für alle Zeitpunkte zwischen den markierten Stellen...
159  for (int i = 0; i < (marker2 - marker1) + 1; i++) {
160  double val = 0;
161  //aktueller Sensordatensatz
162  SensorData* sd = &obj->getSensorDataList()->at(
163  obj->getCurrentSensorIndex());
164 
165  //Summe aller Temperaturmesswerte
166  for (int k = 0; k < int(sd->data.at(i).size()); k++) {
167  val += sd->data.at(i).at(k).temperature;
168  }
169 
170  //neues Temperaturmaximum?
171  if (val > max_energy) {
172  max_energy = val;
173  max_index = marker1 + i;
174  }
175  }
176  } else {
177  //Struktur zum speichern der Analysedaten
178  Analyzer::AnalyzerData_object data;
179  Analyzer analyzer;
180 
181  //aktueller Sensordatensatz
182  SensorData* sd = &obj->getSensorDataList()->at(
183  obj->getCurrentSensorIndex());
184 
185  //speichern des aktuell gewählten Zeitpunktes für die Temperaturverteilung
186  int original_time_index = sd->current_time_index;
187 
188  //für alle Zeitpunkte zwischen den markierten Stellen...
189  for (int i = 0; i < (marker2 - marker1) + 1; i++) {
190  //Berechnen der Temperaturverteilung und Analyse des Objekts zu diesem Zeitpunkt
191  sd->current_time_index = marker1 + i;
192  analyzer.analyzeObject(obj, &data, false,
193  obj->getCurrentSensorIndex());
194 
195  //neues Energiemaximum?
196  if (data.data_sets.at(0).heat_energy > max_energy) {
197  max_energy = data.data_sets.at(0).heat_energy;
198  max_index = marker1 + i;
199  }
200  }
201 
202  //zurücksetzen des Objekts auf den vorher ausgewählten Zeitpunkt
203  sd->current_time_index = original_time_index;
204  obj->calculateIO();
205  }
206 
207  //auswählen des gefundenen Zeitpunkts
208  setValue(max_index);
209 }

211 int GUITimeline::calcStepWidth() {
212 
213  //Zehnerpotenz des beim aktuellen Zoom dargestellten Bereichs (zoom * maxvalue)
214  int to_ten = int(pow(10,int(log10(maxvalue * zoom))));
215  //faktor zur Verfeinerung der kleinsten Skaleneinteilung
216  float refine_factor = 1;
217  //Device Context zum ermitteln zur ausdehnung des dargestellten Texts
218  wxPaintDC dc(this);
219  //Pixel pro Skaleneinteilung
220  float pixelsperstep = (float) GetSize().x / ((maxvalue - minvalue) * zoom);
221 
222  //Versuchen, die Einteilung weiter zu verfeinern, bis sich Text überschneidet
223  for (int i = 0; i < SCALE_REFINE_STEPS; i++) {
224 
225  //zu testende Schrittweite
226  int stepwidth = to_ten * refine_factors[i];
227  //Verschiebung der Ansicht bezogen auf den Zoomfaktor
228  int viewstart = int(delta_v_view / zoom);
229  //die am weitesten links liegende, sichtbare Einteilung
230  int start_index = -int(viewstart / pixelsperstep / stepwidth);
231 
232  //die am weitesten rechts liegende, sichtbare Einteilung
233  int curr_max = start_index
234  + 1. / pixelsperstep * GetSize().x / stepwidth;
235  curr_max = curr_max * stepwidth;
236 
237  wxString str = floattowxstr(curr_max);
238  //Ausdehnung der Beschriftung für die am weitesten rechts liegende, sichtbare Einteilung
239  int text_extend = dc.GetTextExtent(str).x;
240 
241  //Ist diese noch lesbar?
242  if (stepwidth * pixelsperstep > text_extend + 2) {
243  //weiter verfeinern
244  refine_factor = refine_factors[i];
245  } else {
246  break;
247  }
248  }
249 
250  //kleinstmögliche gut lesbare Skaleneinteilung
251  return to_ten * refine_factor;
252 }

263  //Wert an der gegebenen Position

307 void GUITimeline::OnPaint(wxPaintEvent&) {
308 
309  //Zeichenkontext
310  wxPaintDC dc(this);
311  SetDoubleBuffered(true);
312  int width = 0;
313  int height = 0;
314  int captionheight = 20;
315  GetSize(&width, &height);
316 
317  //Pixel pro Zeitpunkt in der Skalendarstellung (bezogen auf die Breite)
318  float pixelsperstep = (float) width / ((maxvalue - minvalue) * zoom);
319  //Schrittweite der Beschriftung
320  float stepwidth = calcStepWidth();
321 
322  //durch verschieben gegebener Start des dargestellten Bereichs
323  int viewstart = int(delta_v_view / zoom);
324  int start_index = -int(viewstart / pixelsperstep / stepwidth);
325 
326  //Zeichnen des Hintergrunds
327  dc.SetPen(wxSystemSettings::GetColour(wxSYS_COLOUR_BACKGROUND));
328  dc.SetBrush(wxSystemSettings::GetColour(wxSYS_COLOUR_BACKGROUND));
329  dc.DrawRectangle(0, 0, width, height);
330  dc.SetBrush(*wxTRANSPARENT_BRUSH);
331  //Zeichnen der Umrandung
332  dc.SetPen(
333  wxPen(wxSystemSettings::GetColour(wxSYS_COLOUR_CAPTIONTEXT), 1,
334  wxSOLID));
335  dc.DrawRectangle(0, 0, width, height);
336 
337  //Zeichnen des nicht auswählbaren Bereichs
338  dc.SetPen(*wxTRANSPARENT_PEN);
339  dc.SetBrush(
340  wxBrush(wxSystemSettings::GetColour(wxSYS_COLOUR_GRAYTEXT),
341  wxSOLID));
342 
343  //nicht auswählbaren Bereich vor dem Minimalwert
344  if (-viewstart < minvalue * pixelsperstep) {
345  dc.DrawRectangle(0, 1, viewstart + minvalue * pixelsperstep, height);
346  }
347  //nicht auswählbaren Bereich nach dem Maximalwert
348  if (viewstart + maxvalue * pixelsperstep - width < 0) {
349  int x = viewstart + maxvalue * pixelsperstep;
350  if (x < 0)
351  x = 0;
352  dc.DrawRectangle(x, 1, width, height);
353  }
354 
355  //Hintergrund der Beschriftung
356  dc.SetBrush(
357  wxBrush(wxSystemSettings::GetColour(wxSYS_COLOUR_3DSHADOW),
358  wxSOLID));
359  dc.SetPen(
360  wxPen(wxSystemSettings::GetColour(wxSYS_COLOUR_3DFACE), 1,
361  wxSOLID));
362  dc.DrawRectangle(0, height - captionheight, width, height);
363 
364  //Beschriften der Zeitleiste
365  dc.SetBackground(*wxTRANSPARENT_BRUSH);
366  dc.SetTextForeground(wxSystemSettings::GetColour(wxSYS_COLOUR_CAPTIONTEXT));
367  dc.SetPen(
368  wxPen(wxSystemSettings::GetColour(wxSYS_COLOUR_BTNTEXT), 1,
369  wxSOLID));
370 
371  //So viele Schritte, wie angezeigt werden können
372  for (int i = start_index;
373  i <= start_index + 1. / pixelsperstep * width / stepwidth; i++) {
374 
375  float x = viewstart + i * stepwidth * pixelsperstep;
376  dc.DrawLine(x, 0, x, height - captionheight);
377  dc.DrawText(floattowxstr(i * stepwidth, maxdigits), x,
378  height - captionheight);
379  }
380 
381  // Sind markierte Stellen angelegt?
382  if (markers != NULL) {
383  dc.SetPen(wxColour(255, 100, 0, 255));
384  //Alle markierten Stellen einzeichnen
385  for (size_t i = 0; i < markers->size(); i++) {
386  dc.DrawLine(viewstart + (minvalue + markers->at(i)) * pixelsperstep,
387  0, viewstart + (minvalue + markers->at(i)) * pixelsperstep,
388  height - captionheight);
389  }
390  } else {
391  cerr << "marker list should not be NULL!" << endl;
392  }
393 
394  //Einzeichnen des aktuellen Zeitpunkts
395  dc.SetPen(*wxGREEN_PEN);
396  dc.DrawLine(viewstart + (float) value * pixelsperstep, 0,
397  viewstart + (float) value * pixelsperstep, height - captionheight);
398 
399  dc.SetTextForeground(*wxGREEN);
400  //Ist eine Namensliste vorhanden?
401  if (names == NULL) {
402  //aktuellen Zeitpunkt mit dem Index beschriften
403  dc.DrawText(floattowxstr(value),
404  viewstart + (float) value * pixelsperstep, 10);
405  } else {
406  //Ist die Namensliste korrekt?
407  if (int(names->size()) < (maxvalue - minvalue)) {
408  cerr << "name vector has wrong size!" << endl;
409  } else {
410  //Beschriften des aktuellen Zeitpunkts mit Index und Namen
411  wxString str = wxString::FromAscii((names->at(value - minvalue)).c_str())
412  + wxT(" (") + floattowxstr(value) + wxT(")");
413 
414  int text_width = dc.GetTextExtent(str).x;
415  int text_x = viewstart+(float)value*pixelsperstep;
416 
417  //Text zu besseren Lesbarkeit verschieben?
418  if (text_x+text_width>width) {
419  text_x -= text_width+10;
420  }
421 
422  dc.DrawText(str,text_x,10);
423  }
424  }
425 }
