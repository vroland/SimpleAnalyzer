88 void GUITimeline::findMaxValue(ObjectData* obj,bool fast) {
89  int val = getValue();
90  int marker1 = getMinValue()-1;
91  int marker2 = getMaxValue()+1;
92  for (size_t i=0;i<markers->size();i++) {
93  int marker = markers->at(i);
94  if (val==marker) {
95  wxMessageBox(wxT("Der Cursor muss sich zwischen zwei Markierungen befinden!"),wxT("Fehler"));
96  return;
97  }
98  if (marker<val && marker>marker1) {
99  marker1 = marker;
100  }
101  if (marker>val && marker<marker2) {
102  marker2 = marker;
103  }
104  }
105  if (marker1 == getMinValue()-1 || marker2 == getMaxValue()+1) {
106  wxMessageBox(wxT("Der Cursor muss sich zwischen zwei Markierungen befinden!"),wxT("Fehler"));
107  return;
108  }
109  float max_energy = -1;
110  float max_index = marker1;
111  if (fast) {
112  for (int i=0;i<(marker2-marker1)+1;i++) {
113  float val = 0;
114  SensorData* sd = &obj->getSensorDataList()->at(obj->getCurrentSensorIndex());
115  for (int k=0;k<int(sd->data.at(i).size());k++) {
116  val+=sd->data.at(i).at(k).temperature;
117  cout << sd->data.at(i).at(k).temperature << endl;
118  }
119  if (val>max_energy) {
120  max_energy = val;
121  max_index = marker1+i;
122  }
123  }
124  } else {
125  Analyzer::AnalyzerData_object data;
126  Analyzer analyzer;
127  for (int i=0;i<(marker2-marker1)+1;i++) {
128  obj->getSensorDataList()->at(obj->getCurrentSensorIndex()).current_time_index = marker1+i;
129  analyzer.analyzeObject(obj,&data,false,obj->getCurrentSensorIndex());
130  if (data.data_sets.at(0).heat_energy>max_energy) {
131  max_energy = data.data_sets.at(0).heat_energy;
132  max_index = marker1+i;
133  }
134  }
135  }
136  setValue(max_index);
137 }
138 int GUITimeline::calcStepWidth() {
139  int to_ten = int(exp10(int(log10(maxvalue*zoom))));
140  float refine_factor = 1;
141  wxPaintDC dc(this);
142  float pixelsperstep = (float)GetSize().x/((maxvalue-minvalue)*zoom);
143  for (int i=0;i<REFINE_STEPS;i++) { // go until Text is unreadable
144  int stepwidth = to_ten*refine_factors[i];
145  int viewstart = int(delta_v_view/zoom);
146  int start_index = -int(viewstart/pixelsperstep/stepwidth);
147 
148  int curr_max = start_index+1./pixelsperstep*GetSize().x/stepwidth;
149  curr_max = curr_max*stepwidth;
150 
151  wxString str = floattowxstr(curr_max);
152  int text_extend = dc.GetTextExtent(str).x;
153  if (stepwidth*pixelsperstep>text_extend+2) {
154  refine_factor = refine_factors[i];
155  } else {
156  break;
157  }
158  }
159 
160  return to_ten*refine_factor;
161 }

196 void GUITimeline::OnPaint(wxPaintEvent&) {
197  int w = 0;
198  int h = 0;
199  GetSize(&w,&h);
200  wxPaintDC dc(this);
201  SetDoubleBuffered(true);
202  // Create graphics context from it
203  //wxGraphicsContext *gc = wxGraphicsContext::Create( dc );
204  int width = 0;
205  int height = 0;
206  int captionheight = 20;
207  GetSize(&width,&height);
208 
209  float pixelsperstep = (float)width/((maxvalue-minvalue)*zoom);
210  float stepwidth = calcStepWidth();
211  int viewstart = int(delta_v_view/zoom);
212  int start_index = -int(viewstart/pixelsperstep/stepwidth);
213  // Clear BG
214  dc.SetPen( wxSystemSettings::GetColour(wxSYS_COLOUR_BACKGROUND) );
215  dc.SetBrush( wxSystemSettings::GetColour(wxSYS_COLOUR_BACKGROUND) );
216  dc.DrawRectangle(0, 0, width, height);
217  dc.SetBrush(*wxTRANSPARENT_BRUSH);
218  // Border
219  dc.SetPen( wxPen(wxSystemSettings::GetColour(wxSYS_COLOUR_CAPTIONTEXT),1,wxSOLID));
220  dc.DrawRectangle(0, 0, width, height);
221 
222  // Unavailable areas
223  dc.SetPen( *wxTRANSPARENT_PEN);
224  dc.SetBrush( wxBrush(wxSystemSettings::GetColour(wxSYS_COLOUR_GRAYTEXT),wxSOLID));
225  if (-viewstart<minvalue*pixelsperstep) {
226  dc.DrawRectangle(0,1,viewstart+minvalue*pixelsperstep,height);
227  }
228  if (viewstart+maxvalue*pixelsperstep-width<0) {
229  int x = viewstart+maxvalue*pixelsperstep;
230  if (x<0) x = 0;
231  dc.DrawRectangle(x,1,width,height);
232  }
233  //Label bg
234  dc.SetBrush( wxBrush(wxSystemSettings::GetColour(wxSYS_COLOUR_3DSHADOW),wxSOLID));
235  dc.SetPen( wxPen(wxSystemSettings::GetColour(wxSYS_COLOUR_3DFACE),1,wxSOLID));
236  dc.DrawRectangle(0,height-captionheight,width,height);
237 
238  dc.SetBackground(*wxTRANSPARENT_BRUSH);
239  dc.SetTextForeground(wxSystemSettings::GetColour(wxSYS_COLOUR_CAPTIONTEXT));
240  dc.SetPen( wxPen(wxSystemSettings::GetColour(wxSYS_COLOUR_BTNTEXT),1,wxSOLID));
241  for (int i=start_index;i<=start_index+1./pixelsperstep*width/stepwidth;i++) {
242  float x = viewstart+i*stepwidth*pixelsperstep;
243  dc.DrawLine(x,0,x,height-captionheight);
244  dc.DrawText(floattowxstr(i*stepwidth,maxdigits),x,height-captionheight);
245  }
246 
247  // Markers
248  if (markers!=NULL) {
249  dc.SetPen( wxColour (255,100,0,255));
250  for (size_t i=0;i<markers->size();i++) {
251  dc.DrawLine(viewstart+(minvalue+markers->at(i))*pixelsperstep,0,viewstart+(minvalue+markers->at(i))*pixelsperstep,height-captionheight);
252  }
253  } else {
254  cerr << "marker list should not be NULL!" << endl;
255  }
256  dc.SetPen( *wxGREEN_PEN);
257  dc.DrawLine(viewstart+(float)value*pixelsperstep,0,viewstart+(float)value*pixelsperstep,height-captionheight);
258  dc.SetTextForeground(*wxGREEN);
259  if (names==NULL) {
260  dc.DrawText(floattowxstr(value),viewstart+(float)value*pixelsperstep,10);
261  } else {
262  if (int(names->size())<(maxvalue-minvalue)) {
263  cerr << "name vector has wrong size!" << endl;
264  } else {
265  wxString str = wxString::FromAscii((names->at(value-minvalue)).c_str())+wxT(" (")+floattowxstr(value)+wxT(")");
266  int text_width = dc.GetTextExtent(str).x;
267  int text_x = viewstart+(float)value*pixelsperstep;
268  if (text_x+text_width>width) {
269  text_x -= text_width+10;
270  }
271  dc.DrawText(str,text_x,10);
272  }
273  }
