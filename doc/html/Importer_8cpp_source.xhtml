51 string getTextBlock(string data, int n) {
52 
53  //Position des abschließenden Leerzeichens für den ersten Block
54  size_t position = data.find(" ");
55  //Speicher für das Ende des vorherigen Blocks
56  size_t previousPos = 0;
57 
58  //Ist der String Leerzeichenlos?
59  if (position == data.npos) {
60  return "";
61  }
62 
63  //Durchlaufen aller Textblöcke bis zum gewünschten Index
64  for (int i = 0; i < n; i++) {
65 
66  //aus Ende des verherigen Blocks den Anfang des aktuellen Berechnen.
67  previousPos = position + 1;
68  //Position des nächsten Blocks suchen
69  position = data.find(" ", position + 1);
70  //Ende der Zeile erreicht?
71  if (position == data.npos) {
72  //Ist der Index außerhalb des möglichen Bereichs?
73  if (n - 1 > i) {
74  return "";
75  }
76  }
77  }
78 
79  //Zurückgeben des aktuellen Textblocks
80  return data.substr(previousPos, position - (previousPos));
81 }

294  tetgenio* io = new tetgenio();
295 
296  //Schreiben der Punkte des Teilobjekts
297  io->firstnumber = 0;
298  io->numberofpoints = currentMatPointCount;
299  io->pointlist = new REAL[currentMatPointCount * 3];
300  for (int i = 0; i < currentMatPointCount * 3; i++) {
301  io->pointlist[i] = points.at(3 * currentMatFirstPoint + i);
302  }
303 
304  //Schreiben der Flächen des Teilobjekts
305  io->numberoffacets = faces.size();
306  io->facetlist = new tetgenio::facet[io->numberoffacets];
307  for (int i = 0; i < io->numberoffacets; i++) {
308  tetgenio::facet* face = &io->facetlist[i];
309  face->numberofpolygons = 1;
310  face->polygonlist =
311  new tetgenio::polygon[face->numberofpolygons];
312  face->numberofholes = 0;

315  vector<int>* subvec = &faces.at(i);
316  poly->numberofvertices = subvec->size();
317  poly->vertexlist = new int[poly->numberofvertices];
318  for (int j = 0; j < poly->numberofvertices; j++) {
319  poly->vertexlist[j] = subvec->at(j) - 1;
320  }
321  }
322 
323  //Verknüpfen der Geometriedaten mit dem Objekteigenschaften
324  data->getMaterials()->at(currentMatIndex).tetgeninput = io;
325  }
326 
327  //mitzählen des Anfangspunktindex für das nächste Objekt
328  currentMatFirstPoint += currentMatPointCount;
329  //Zurücksetzen der Punktanzahl
330  currentMatPointCount = 0;
331  //Zurücksetzen der Eckpunktindexliste für die Flächen
332  faces.clear();
333  continue;
334  }
335  }
336 
337  file.close();
338 
339  //Objekt und Materialdaten erfolgreich geladen
340  return ObjectData::OD_SUCCESS;
