272 void GUIMainWindow::OnSDTLPrevMarker(wxCommandEvent &event) {

274  if (markers->size()==0) {
275  wxMessageBox(wxT("Es sind keine Analyze-Marker gesetzt!"),wxT("Hinweis"),wxICON_INFORMATION);

287  propbox->getSdTimeline()->setValue(markers->at(i));

470 void GUIMainWindow::OnMenuImportObj(wxCommandEvent &event)
471 {
472  wxFileDialog *OpenDialog= new wxFileDialog(this, wxT("Datei öffnen..."), _(""), _(""), _("Wavefront-Objektdateien (*.obj)|*.obj"), wxFD_OPEN|wxFD_FILE_MUST_EXIST);
473  if ( OpenDialog->ShowModal() == wxID_OK )
474  {
475  ObjectData* newobj = new ObjectData();
476  wxString path = OpenDialog->GetPath();
477  int status = newobj->loadFromFile(path);
478  switch (status) {
479  case ObjectData::OD_LOAD_ALREADY_LOADED:
480  wxMessageBox(path+wxT(" ist bereits geöffnet!"));
481  delete newobj;
482  break;
483  case ObjectData::OD_LOAD_INVALID_FILE:
484  wxMessageBox(path+wxT(" ist keine Wavefront(.obj)-Datei."));
485  delete newobj;
486  break;
487  case ObjectData::OD_LOAD_INVALID_SENSOR_FILE:
488  wxMessageBox(wxT("Sensordaten konnten nicht geladen werden: ")+(path.BeforeLast('.')+wxT(".(t)sd")));
489  delete newobj;
490  break;
491  case ObjectData::OD_SUCCESS:
492  addObject(newobj);
493  break;
494  default:
495  cout << "undefined status value!" << endl;
496  delete newobj;
497  break;
498  }
499  propbox->setCurrentMaterial(0);
500  updateObjectPropGUI();
501  updateViewPropGUI();
502  }
503  OpenDialog->Close();
504  OpenDialog->Destroy();
505  propbox->getUpToDateLbl()->Hide();
506 }
