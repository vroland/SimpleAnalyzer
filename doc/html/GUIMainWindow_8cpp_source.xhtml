420 void GUIMainWindow::OnSDTLNextMarker(wxCommandEvent &event) {
421 
422  //indices aller markierten Zeitpunkte, ist aufsteigend sortiert
423  vector<int>* markers = propbox->getSdTimeline()->getMarkers();
424 
425  //Sind marker gesetzt?
426  if (markers->size() == 0) {
427  wxMessageBox(wxT("Es sind keine Analyze-Marker gesetzt!"),
428  wxT("Hinweis"), wxICON_INFORMATION);
429  } else {
430  //nächsten marker suchen
431  for (size_t i = 0; i < markers->size(); i++) {
432  if (markers->at(i) > propbox->getSdTimeline()->getValue()) {
433  propbox->getSdTimeline()->setValue(markers->at(i));
434  break;
435  }
436  }

440 void GUIMainWindow::OnSDTLPrevMarker(wxCommandEvent &event) {
441 
442  //indices aller markierten Zeitpunkte, ist aufsteigend sortiert
443  vector<int>* markers = propbox->getSdTimeline()->getMarkers();
444 
445  //Sind marker gesetzt?
446  if (markers->size() == 0) {
447  wxMessageBox(wxT("Es sind keine Analyze-Marker gesetzt!"),
448  wxT("Hinweis"), wxICON_INFORMATION);
449  } else {
450  //vorherigen marker suchen
451  for (size_t i = 0; i < markers->size(); i++) {
452 
453  //ist die markierte Stelle hinter/an dem aktuell ausgewählte Zeitpunkt?
454  if (markers->at(i) >= propbox->getSdTimeline()->getValue()) {
455  //gibt es eine merkierte Stelle davor?
456  if (i > 0) {
457  //vorherige Stelle verwenden
458  propbox->getSdTimeline()->setValue(markers->at(i - 1));
459  } else {
460  //erste Stelle verwenden
461  propbox->getSdTimeline()->setValue(markers->at(0));
462  }
463 
464  break;
465  }
466 
467  //Sind alle markiereten Stellen vor dem aktuellen Zeitpunkt?
468  if (i == markers->size() - 1) {
469  //die naheste verwenden
470  propbox->getSdTimeline()->setValue(markers->at(i));
471  }
472  }
473  }
474 }

784  wxFileDialog *OpenDialog = new wxFileDialog(this, wxT("Datei öffnen..."),
785  _(""), _(""), _("Wavefront-Objektdateien (*.obj)|*.obj"),
786  wxFD_OPEN | wxFD_FILE_MUST_EXIST);
787 
788  //Öffnen bestätigt?
789  if (OpenDialog->ShowModal() == wxID_OK) {
790 
791  //Versuch, ein neues Objekt zu Laden
792  ObjectData* newobj = new ObjectData();
793  wxString path = OpenDialog->GetPath();
794  int status = newobj->loadFromFile(path);
795 
796  //Fehler auswerten
797  switch (status) {
798  case ObjectData::OD_LOAD_ALREADY_LOADED:
799  wxMessageBox(path + wxT(" ist bereits geöffnet!"));
800  delete newobj;
801  break;
802  case ObjectData::OD_LOAD_INVALID_FILE:
803  wxMessageBox(path + wxT(" ist keine Wavefront(.obj)-Datei."));
804  delete newobj;
805  break;
806  case ObjectData::OD_LOAD_INVALID_SENSOR_FILE:
807  wxMessageBox(
808  wxT("Sensordaten konnten nicht geladen werden: ")
809  + (path.BeforeLast('.') + wxT(".(t)sd")));
810  delete newobj;
811  break;
812  case ObjectData::OD_SUCCESS:
813  //Objekt im Propgramm aufnehmen
814  addObject(newobj);
815  break;
816  default:
817  cerr << "undefined status value!" << endl;
818  delete newobj;
819  break;
820  }
821 
822  //aktualisieren der Oberfläche
823  propbox->setCurrentMaterial(0);
824  updateObjectPropGUI();
825  updateViewPropGUI();
826  propbox->getUpToDateLbl()->Hide();
827  }

829  OpenDialog->Close();
830  OpenDialog->Destroy();
