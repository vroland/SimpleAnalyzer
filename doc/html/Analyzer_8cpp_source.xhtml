26 void Analyzer::analyzeObject(ObjectData* obj, AnalyzerData_object* out,
27  bool use_markers, int sdindex) {
28 
29  out->volume = 0;
30  //aktuell ausgewählte Sensordaten, um die Auswahl nach der Analyse wieder zurück zu setzen
31  int original_sd_index = obj->getCurrentSensorIndex();
32  //hat sich der ausgewählte Zeitpunkt der Sensordaten verändert?
33  bool sd_time_changed = false;
34  //objekt zum Berechnen der Temperatuverteilung
35  MeshProcessor processor;
36 
37  //für alle Sensordaten des Objekt bzw. für den evtl. festgelegten Sensordatensatz
38  for (unsigned int s = (sdindex > -1) ? sdindex : 0;
39  s < ((sdindex > -1) ? sdindex + 1 : obj->getSensorDataList()->size());
40  s++) {
41 
42  //der aktualle Sensordatensatz
43  obj->setCurrentSensorIndex(s);
44  SensorData* sd = &obj->getSensorDataList()->at(s);
45 
46  //Anzahl der zu analysierende Unterdatensätze
47  int subsets = sd->timed ? sd->markers.size() : 1;
48  subsets = (sd->timed && !use_markers) ? 1 : subsets;
49 
50  //Anzahl der bisher verarbeiteten Unterdatensätze
51  int prev_set_count = out->data_sets.size();
52  //Platz für die neuen Unterdatensätze
53  out->data_sets.resize(prev_set_count + subsets);
54 
55  //aktuell ausgewählter Zeitpunkt des Datensatzes
56  int original_time_index = sd->current_time_index;
57 
58  //Für alle Unterdatensätze
59  for (int ts = 0; ts < subsets; ts++) {
60  out->volume = 0;
61 
62  //Unterdatensatz als aktiv auswählen
63  sd->current_time_index = sd->timed ? sd->markers.at(ts) : 0;
64  sd->current_time_index =
65  (sd->timed && !use_markers) ? 0 : sd->current_time_index;
66  //Temperaturverteilung berechnen
67  processor.process(obj);
68 
69  //Struktur zum Speichern der Analysedaten
70  AnalyzerData_dataset* data_set = &out->data_sets.at(
71  prev_set_count + ts);
72  data_set->name =
73  sd->timed ?
74  sd->subnames.at(sd->current_time_index) : sd->name;
75  data_set->heat_energy = 0;
76 
77  data_set->mat_data.resize(obj->getMaterials()->size());
78 
79  //Für alle Materialien...
80  for (unsigned int i = 0; i < obj->getMaterials()->size(); i++) {
81 
82  AnalyzerData_material* mat_data = &data_set->mat_data.at(i);
83  ObjectData::MaterialData* mat = &obj->getMaterials()->at(i);
84  mat_data->name = mat->name;
85  //Gesantvolumen des Materials
86  double matvolume = 0;
87  //Gesantenergie des Materials
88  double matenergy = 0;
89 
90  //für alle Tetraeder...
91  for (int j = 0; j < mat->tetgenoutput->numberoftetrahedra; j++) {
92  int* indices = &mat->tetgenoutput->tetrahedronlist[4 * j];
93  Vector3D v1 = Vector3D(
94  &mat->tetgenoutput->pointlist[3 * indices[0]]);
95  Vector3D v2 = Vector3D(
96  &mat->tetgenoutput->pointlist[3 * indices[1]]);
97  Vector3D v3 = Vector3D(
98  &mat->tetgenoutput->pointlist[3 * indices[2]]);
99  Vector3D v4 = Vector3D(
100  &mat->tetgenoutput->pointlist[3 * indices[3]]);
101  v2.sub(&v1);
102  v3.sub(&v1);
103  v4.sub(&v1);
104  Vector3D* vvec = v2.crossProduct(&v3);
105  //Volumen des Tetraeders berechnen
106  double volume = 1. / 6. * abs(vvec->dotProduct(&v4));
107  matvolume += volume;
108  delete vvec;
109 
110  //Durchschnittstemperatur des Tetraeders ermitteln
111  double averagetemp = 0;
112  for (int k = 0; k < 4; k++) {
113  averagetemp +=
114  mat->tetgenoutput->pointattributelist[mat->tetgenoutput->numberofpointattributes
115  * indices[k]];
116  }
117  averagetemp /= 4.;
118 
119  //Energie des Tetraeders ausgehend von der Durchschnittstemperatur
120  matenergy += averagetemp * volume * mat->density
121  * mat->specificheatcapacity;
122 
123  }
124 
125  mat_data->volume = matvolume;
126  mat_data->heat_energy = matenergy;
127 
128  //Gesamtvolumen mitrechnen
129  out->volume += matvolume;
130  //Energie des Datensatzes mitrechnen
131  data_set->heat_energy += matenergy;
132  }
133  }
134 
135  //Muss der ausgewählte Zeitpunkt zurück gesetzt werden?
136  sd_time_changed |= (sd->current_time_index != original_time_index);
137  sd->current_time_index = original_time_index;
138  }
139 
140  //Muss die Temepraturverteilung des Objekts zum Wiederherstellen des Ausgangszustands neu berechnet werden?
141  if (obj->getCurrentSensorIndex() != original_sd_index || sd_time_changed) {
142  obj->setCurrentSensorIndex(original_sd_index);
143  processor.process(obj);
144  }
145 }
