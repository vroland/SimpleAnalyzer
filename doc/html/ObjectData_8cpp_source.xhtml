28 int ObjectData::loadFromFile(wxString &path) {
29  if (materials.size()!=0) {
30  return OD_LOAD_ALREADY_LOADED;
31  }
32  wxString end = _T("obj");
33  wxString path_without_name = path.BeforeLast('.');
34  if (path.AfterLast('.')!=end) {
35  return OD_LOAD_INVALID_FILE;
36  }
37  setlocale(LC_NUMERIC, "C");
38  Importer importer;
39  importer.ImportObj((path_without_name+_T(".obj")).ToAscii(),this);
40  int status = importer.LoadSensorData((path_without_name+_T(".sd")).ToAscii(),this);
41  if (status==OD_FAILURE) {
42  int status = importer.LoadTimedData((path_without_name+_T(".tsd")).ToAscii(),this);
43  if (status==OD_FAILURE) {
44  return OD_LOAD_INVALID_SENSOR_FILE;
45  }
46  }
47  wxString wxfn = path_without_name.AfterLast(PATH_SEPARATOR);
48  string oname = string(wxfn.ToAscii());
49  string tname = oname;
50  bool unique = false;
51  int count = 0;
52  while (!unique) {
53  bool found = false;
54  for (unsigned int i=0;i<data_objects.size();i++) {
55  if (data_objects.at(i)->name==tname) {
56  found = true;
57  }
58  }
59  if (!found) {
60  unique = true;
61  } else {
62  count++;
63  stringstream ss (stringstream::in | stringstream::out);
64  ss <<oname;
65  ss <<" ("<<count<<")";
66  tname =ss.str();
67  }
68  }
69  name = tname;
70  calculateIO();
71  return OD_SUCCESS;
72 }

83 int ObjectData::calculateIO() {
84  setlocale(LC_NUMERIC, "C");
85  for (unsigned int i=0;i<materials.size();i++) {
86  MaterialData* data = &materials.at(i);
87  if (data->tetgenoutput!=NULL) {
88  delete data->tetgenoutput;
89  }
90  if (quality==0) {
91  cerr << "Error: quality is 0! Are object properties set properly?" << endl;
92  return OD_FAILURE;
93  }
94  data->tetgenoutput = new tetgenio();
95  stringstream ss (stringstream::in | stringstream::out);
96  ss <<"Qpq1.414a";
97  ss << maxvolume;
98  string args =ss.str();//"Qpq1.414a0.001";
99  tetrahedralize(const_cast<char*> (args.c_str()), data->tetgeninput, data->tetgenoutput,NULL,NULL);
100  cout << data->tetgenoutput->numberoftrifaces << endl;
101  }
102  MeshProcessor meshprocessor;
103  meshprocessor.process(this);
