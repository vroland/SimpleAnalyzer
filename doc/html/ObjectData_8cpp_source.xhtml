33 int ObjectData::loadFromFile(wxString &path) {
34 
35  //sind schon Daten mit dem Objekt verknüpft?
36  if (materials.size() != 0) {
37  return OD_LOAD_ALREADY_LOADED;
38  }
39 
40  //Ist die angegebene Datei eine .obj-Datei?
41  wxString end = _T("obj");
42  wxString path_without_name = path.BeforeLast('.');
43  if (path.AfterLast('.') != end) {
44  return OD_LOAD_INVALID_FILE;
45  }
46 
47  //Dezimaltrennzeichen auf Punkt setzen
48  setlocale(LC_NUMERIC, "C");
49 
50  //Objekt importieren
51  Importer importer;
52  importer.ImportObj(string((path_without_name + _T(".obj")).ToUTF8().data()),
53  this);
54 
55  //Sensordaten importieren
56  int status = importer.LoadSensorData(
57  string((path_without_name + _T(".sd")).ToUTF8().data()), this);
58  //Fehlschlag? -> Versuchen, zeitgesteuerte Sensordaten zu importieren
59  if (status == OD_FAILURE) {
60  int status = importer.LoadTimedData(
61  string((path_without_name + _T(".tsd")).ToUTF8().data()), this);
62  if (status == OD_FAILURE) {
63  //Sensordaten konnten nicht geladen werden
64  return OD_LOAD_INVALID_SENSOR_FILE;
65  }
66  }
67 
68  //Namen für das Objekt aus dem Dateinamen ermitteln
69  wxString wx_filename = path_without_name.AfterLast(PATH_SEPARATOR);
70  string obj_name = string(wx_filename.ToAscii());
71  string temp_name = obj_name;
72 
73  bool unique = false;
74  //Anzahl der Durchläufe
75  int count = 0;
76  //Solange ein Objekt mit dem selben Namen existiert...
77  while (!unique) {
78 
79  //Existiert bereits ein Objekt mit diesem Namen?
80  bool found = false;
81  for (unsigned int i = 0; i < wxGetApp().getDataObjects()->size(); i++) {
82  if (wxGetApp().getDataObjects()->at(i)->name == temp_name) {
83  found = true;
84  }
85  }
86 
87  //Objektnamen einzigartig?
88  if (!found) {
89  unique = true;
90  } else {
91  //Objekt eine Nummer zu Identifisierung an den Namen geben
92  count++;
93  stringstream ss(stringstream::in | stringstream::out);
94  ss << obj_name;
95  ss << " (" << count << ")";
96  temp_name = ss.str();
97  }
98  }
99 
100  name = temp_name;
101 
102  //Geometrie und Temperaturverteilung für das Objekt berechnen
103  calculateIO();
104 
105  return OD_SUCCESS;
106 }

120 int ObjectData::calculateIO() {
121 
122  //Dezimaltrennzeichen auf Punkt setzen
123  setlocale(LC_NUMERIC, "C");
124 
125  //Für alle Materialien...
126  for (unsigned int i = 0; i < materials.size(); i++) {
127  MaterialData* data = &materials.at(i);
128 
129  //Muss erst alte Geometrie gelöscht werden?
130  if (data->tetgenoutput != NULL) {
131  delete data->tetgenoutput;
132  }
133 
134  //Objekteigenschaften gültig?
135  if (quality == 0) {
136  cerr << "Error: quality is 0! Are object properties set properly?"
137  << endl;
138  return OD_FAILURE;
139  }
140 
141  //Speicherstruktur für die neue Geometrie erzeugen
142  data->tetgenoutput = new tetgenio();
143 
144  //Die neue Geometrie mit tetgen erzeugen
145  stringstream ss(stringstream::in | stringstream::out);
146  ss << "Qpq1.414a";
147  ss << maxvolume;
148  string args = ss.str(); //"Qpq1.414a0.001";
149  tetrahedralize(const_cast<char*>(args.c_str()), data->tetgeninput,
150  data->tetgenoutput, NULL, NULL);
151  }
152 
153  //Die Temperaturverteilung für die neue Geometrie berechnen
154  MeshProcessor meshprocessor;
155  meshprocessor.process(this);
156 
157  return OD_SUCCESS;
158 }
