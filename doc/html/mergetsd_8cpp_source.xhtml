40  string getTextBlock(string data, int n) {
41 
42  //Position des abschließenden Leerzeichens für den ersten Block
43  size_t position = data.find(" ");
44  //Speicher für das Ende des vorherigen Blocks
45  size_t previousPos = 0;
46 
47  //Ist der String Leerzeichenlos?
48  if (position == data.npos) {
49  return "";
50  }
51 
52  //Durchlaufen aller Textblöcke bis zum gewünschten Index
53  for (int i = 0; i < n; i++) {
54 
55  //aus Ende des verherigen Blocks den Anfang des aktuellen Berechnen.
56  previousPos = position + 1;
57  //Position des nächsten Blocks suchen
58  position = data.find(" ", position + 1);
59  //Ende der Zeile erreicht?
60  if (position == data.npos) {
61  //Ist der Index außerhalb des möglichen Bereichs?
62  if (n - 1 > i) {
63  return "";
64  }
65  }
66  }
67 
68  //Zurückgeben des aktuellen Textblocks
69  return data.substr(previousPos, position - (previousPos));
70  }
