70 void GUIGLCanvas::OnMouseMove(wxMouseEvent &event) {
71  bool refresh = false;
72  if (prev_mouse_x>-1) {
73  if (event.m_leftDown) {
74  renderer.getViewport()->rotationX +=event.m_x-prev_mouse_x;
75  renderer.getViewport()->rotationY +=event.m_y-prev_mouse_y;
76  refresh = true;
77  }
78  if (event.m_middleDown) {
79  Vector3D invec = Vector3D((event.m_x-prev_mouse_x)*.1,-(event.m_y-prev_mouse_y)*.1,0);
80  Matrix3D view;
81  view.rotateX(renderer.getViewport()->rotationY*M_PI/180.);
82  view.rotateY(renderer.getViewport()->rotationX*M_PI/180.);
83  view.transpose();
84  Vector3D* outvec = view.mult(&invec);
85 
86  renderer.getViewport()->cameraPosition->add(outvec);
87  delete outvec;
88  refresh = true;
89  }
90  }
91  prev_mouse_x = event.m_x;
92  prev_mouse_y = event.m_y;
93  if (refresh) {
94  Refresh(false,NULL);
95  }
