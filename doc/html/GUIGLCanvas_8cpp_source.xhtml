105 void GUIGLCanvas::OnMouseMove(wxMouseEvent &event) {
106 
107  //Aktualisierung nötig?
108  bool refresh = false;
109 
110  //Mausstatus initialisiert?
111  if (prev_mouse_x > -1) {
112 
113  //Kamera drehen?
114  if (event.m_leftDown) {
115  renderer.getViewport()->rotationX += event.m_x - prev_mouse_x;
116  renderer.getViewport()->rotationY += event.m_y - prev_mouse_y;
117  refresh = true;
118  }
119 
120  //Kamera verschieben?
121  if (event.m_middleDown) {
122  //Verschiebungsvektor
123  Vector3D invec = Vector3D((event.m_x - prev_mouse_x) * .1,
124  -(event.m_y - prev_mouse_y) * .1, 0);
125  //Matrix zum Transformieren des Verschiebungsvektors durch die Kameradrehung
126  Matrix3D view;
127  view.rotateX(renderer.getViewport()->rotationY * M_PI / 180.);
128  view.rotateY(renderer.getViewport()->rotationX * M_PI / 180.);
129  view.transpose();
130  //Transformieren des Verschiebungsvektors
131  Vector3D* outvec = view.mult(&invec);
132 
133  //Verschieben der Kamera
134  renderer.getViewport()->cameraPosition->add(outvec);
135  delete outvec;
136  refresh = true;
137  }
138  }
139 
140  //Speichern des alten Mausstatus
141  prev_mouse_x = event.m_x;
142  prev_mouse_y = event.m_y;
143 
144  //Aktualisierung wenn nötig
145  if (refresh) {
146  Refresh(false, NULL);
147  }
148 }
