89  sdTimeline->Hide();
90  analyzeMarkerCheckBox->Hide();

92  nextMarkerBt->Hide();
93  prevMarkerBt->Hide();
94  findMaxBt->Hide();
95  int sdheight = 0;
96 
97  //Ist ein Objekt geladen?
98  if (wxGetApp().getCurrentDataObjectIndex() > -1) {
99  ObjectData* object = wxGetApp().getActiveObject();
100  SensorData* sd = &object->getSensorDataList()->at(
101  sensorDataList->GetSelection());
102 
103  //Sind die verwendeten Sensordaten zeitbezogen?
104  if (sd->timed) {
105  sdTimeline->setMarkerList(&sd->markers);
106  sdTimeline->SetSize(x + 20, y + 250, 260, 80, 0);
107  sdTimeline->setValue(sd->current_time_index);
108  sdTimeline->setMinValue(0);
109  sdTimeline->setMaxValue(sd->timestamps.size() - 1);
110  sdTimeline->setNameList(&sd->subnames);
111  sdTimeline->Show();
112  sdheight = 120;
113 
114  analyzeMarkerCheckBox->Show();
115  clearAnalyzeMarkerBt->Show();
116  nextMarkerBt->Show();
117  prevMarkerBt->Show();
118  findMaxBt->Show();
119  prevMarkerBt->SetSize(x + 20, y + 330, 25, 25, 0);
120  nextMarkerBt->SetSize(x + 45, y + 330, 25, 25, 0);
121  findMaxBt->SetSize(x + 70, y + 330, 40, 25, 0);
122  analyzeMarkerCheckBox->SetSize(x + 110, y + 330, 110, 25, 0);
123  clearAnalyzeMarkerBt->SetSize(x + 220, y + 330, 60, 25, 0);
124  }
125  }
126 
127  matListBoxLbl->SetSize(x + 10, y + 240 + sdheight, 300, 20, 0);
128  matListBox->SetSize(x + 10, y + 260 + sdheight, 270, 60, 0);
129  matPropBox->SetSize(x + 10, y + 330 + sdheight, 270, 195, 0);
130  matNameLbl->SetSize(x + 20, y + 350 + sdheight, 300, 20, 0);
