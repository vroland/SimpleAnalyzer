56 void GUIAnalyzePointWindow::analyzePoint(wxCommandEvent &event) {
57 
58  Analyzer analyzer;

60  Analyzer::AnalyzerData_point data;

63  interpolator.setMode(
64  (Interpolator::InterpolationMode) interpolationModeList->GetSelection());
65 
66  //Setzend des Dezimaltrennzeichens auf Punkt
67  setlocale(LC_NUMERIC, "C");
68 
69  //Auslesen der Koordinaten des gesuchten Punkts aus der Oberfläche
70  Vector3D point = Vector3D(atof(xedit->GetValue().ToAscii()),
71  atof(yedit->GetValue().ToAscii()),
72  atof(zedit->GetValue().ToAscii()));
73 
74  //Analysieren des Punktes
75  analyzer.analyzePoint(wxGetApp().getActiveObject(), &point, &data, &interpolator);
76 
77  //Sammeln der gewonnenen Informationen als Text
78  wxString message = wxT("Analysedaten des Punktes (");
79  message.Append(floattowxstr(point.getX()));
80  message.Append(wxT("|"));
81  message.Append(floattowxstr(point.getY()));
82  message.Append(wxT("|"));
83  message.Append(floattowxstr(point.getZ()));
84  message.Append(wxT("):\nTemperatur: "));
85  message.Append(floattowxstr(data.value));
86  message.Append(wxT("\nExtrapoliert: "));
87  data.extrapolated ? message.Append(wxT("ja")) : message.Append(wxT("nein"));
88 
89  //Ausgeben der Informationen als Nachricht
90  wxMessageBox(message, wxT("Punktanalyse"), wxICON_INFORMATION);
