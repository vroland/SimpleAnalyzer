65 void GUIRenderCutCanvas::OnMouseMove(wxMouseEvent &event) {
66 
67  int width = 0;
68  int height = 0;
69  GetSize(&width, &height);
70 
71  //Position der Temperaturverteilungsgrafik auf der Zeichenfläche
72  wxPoint img_point(width / 2 + deltaX - int(image->GetWidth() * zoom) / 2,
73  height / 2 + deltaY - int(image->GetHeight() * zoom) / 2);
74  //Größe der Temperaturverteilungsgrafik
75  wxPoint img_dim(image->GetWidth(), image->GetHeight());
76 
77  //Müssen die Mauseingaben zur Skala weitergeleitet werden?
78  if (mouse_to_scalepanel) {
79  scalepanel->handleMouse(event, img_point, img_dim, zoom);
80  } else {
81  //anpassen der Größe und Position der Grafik and die Temperaturverteilungsgrafik
82  scalepanel->fitBounds(img_dim, false);
83 
84  //Verschieben der Ansicht auf die Grafik?
85  if (event.m_leftDown) {
86  deltaX += (event.m_x - current_mx);
87  deltaY += (event.m_y - current_my);
88  }
89  }
90 
91  //Zurücksetzen der Umleitung der Mauseingaben nach loslassen der Maustaste
92  if (!event.m_leftDown && mouse_to_scalepanel) {
93  mouse_to_scalepanel = false;
94  }
95 
96  //Speichern der Mausinformationen zum ermitteln der Bewegung
97  current_mx = event.m_x;
98  current_my = event.m_y;
99 
100  Refresh();
101  SetFocus();
102 }

130 void GUIRenderCutCanvas::onCanvasPaint(wxPaintEvent &event) {
131 
132  //Device Kontext zum zeichnen erstellen
133  wxPaintDC dc(this);
134 
135  int width = 0;
136  int height = 0;
137  GetSize(&width, &height);
138  int imgwidth = 0;
139  int imgheight = 0;
140 
141  //Ist ein Bild mit der Zeichenfläche verknüpft?
142  if (image != NULL) {
143  imgwidth = image->GetWidth();
144  imgheight = image->GetHeight();
145  }
146 
147  //Durch den Zoomfaktor veränderte Bildmaße
148  int zwidth = int(imgwidth * zoom);
149  int zheight = int(imgheight * zoom);
150  //Hintergrund löschen
151  dc.SetPen(*wxGREY_PEN);
152  dc.SetBrush(*wxGREY_BRUSH);
153  dc.DrawRectangle(0, 0, width, height);
154  //Position für die Grafik auf der Zeichenfläche errechnen
155  int imgx = width / 2 + deltaX - zwidth / 2;
156  int imgy = height / 2 + deltaY - zheight / 2;
157 
158  //Ist ein Bild mit der Zeichenfläche verknüpft?
159  if (image != NULL) {
160  //Bild zum zeichnen kopieren und skalieren
161  wxImage drawimg = image->Copy();
162  drawimg.Rescale(zwidth, zheight, wxIMAGE_QUALITY_NORMAL);
163  //Bild zeichnen
164  dc.DrawBitmap(wxBitmap(drawimg), imgx, imgy);
165 
166  //Rahmen für das Bild zeichnen
167  dc.SetPen(*wxBLACK);
168  int bx = width / 2 + deltaX - zwidth / 2;
169  int by = height / 2 + deltaY - zheight / 2;
170  dc.DrawLine(bx, by, bx + zwidth, by);
171  dc.DrawLine(bx, by, bx, by + zheight);
172  dc.DrawLine(bx, by + zheight, bx + zwidth, by + zheight);
173  dc.DrawLine(bx + zwidth, by, bx + zwidth, by + zheight);
174  }
175 
176  //zeichnen der Skala
177  wxPoint img_pos(imgx, imgy);
178  scalepanel->paintTo(dc, zoom, img_pos);
179 
180  //Bild für den Hintergrund der Statusleiste
181  wxImage statusimg = wxImage(width, 30, true);
182  statusimg.InitAlpha();
183 
184  //Hintergrund der Statusleiste halbtransparent machen
185  for (int i = 0; i < 30 * width; i++) {
186  statusimg.SetAlpha(i % width, i / width, 150);
187  }
188 
189  //zeichnen des Hintergrunds für die Statusleiste
190  wxBitmap statusbmp(statusimg);
191  dc.DrawBitmap(statusbmp, 0, height - 30);
192 
193  /*
194  * zeichnen der Beschriftung der Statusleiste
195  */
196 
197  dc.SetTextForeground(wxColour(255, 255, 255));
198 
199  //befindet sich der Mauszeiger über dem Bild?
200  if (current_mx >= imgx && current_mx < imgx + zwidth && current_my >= imgy
201  && current_my < imgy + zheight) {
202 
203  //die dem Mauszeiger entsprechende Position auf der Grafik
204  int x = int((current_mx - imgx) / zoom);
205  int y = int((current_my - imgy) / zoom);
206 
207  //der Temperaturwert an der Stelle des Mauszeigers
208  float value = 0;
209 
210  //ist eine Temperaturverteilung erstellt?
211  if (value_img != NULL) {
212  value = value_img[y * imgwidth + x];
213  }
214 
215  //Ausgeben der Mausposition und der Temperatur in der Statusleiste
216  wxString val_str = floattowxstr(value) + wxT("°C");
217  if (abs(value + 300) < .0001) {
218  val_str = wxT(" - ");
219  }
220  value = int(value * 100.) / 100.;
221  dc.DrawText(wxT("X:") + floattowxstr(x) + wxT(" Y: ") + floattowxstr(y)
222  + wxT(" Wert: ") + val_str, 20, height - 25);
223  }
