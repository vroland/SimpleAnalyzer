49 void GUIRenderCutCanvas::OnMouseMove(wxMouseEvent &event) {
50  int width = 0;
51  int height = 0;
52  GetSize(&width,&height);
53  wxPoint img_point(width/2+deltaX-int(image->GetWidth()*zoom)/2,
54  height/2+deltaY-int(image->GetHeight()*zoom)/2);
55  wxPoint img_dim(image->GetWidth(),image->GetHeight());
56  if (mouse_to_scalepanel) {
57  scalepanel->handleMouse(event,img_point,img_dim,zoom);
58  } else {
59  scalepanel->fitBounds(img_dim,false);
60  if (event.m_leftDown) {
61  deltaX+=(event.m_x-current_mx);
62  deltaY+=(event.m_y-current_my);
63  //original_x = event.m_x;
64  //original_y = event.m_y;
65  }
66  }
67  if (!event.m_leftDown && mouse_to_scalepanel) {
68  mouse_to_scalepanel = false;
69  }
70  current_mx = event.m_x;
71  current_my = event.m_y;
72  Refresh();
73 }

91 void GUIRenderCutCanvas::onCanvasPaint(wxPaintEvent &event) {
92  // Create paint DC
93  wxPaintDC dc(this);
94  // Create graphics context from it
95  //wxGraphicsContext *gc = wxGraphicsContext::Create( dc );
96  int width = 0;
97  int height = 0;
98  int imgwidth = 0;
99  int imgheight = 0;
100  if (image!=NULL) {
101  imgwidth = image->GetWidth();
102  imgheight = image->GetHeight();
103  }
104  GetSize(&width,&height);
105  int zwidth = int(imgwidth*zoom);
106  int zheight = int(imgheight*zoom);
107  // Clear BG
108  dc.SetPen( *wxGREY_PEN );
109  dc.SetBrush( *wxGREY_BRUSH);
110  dc.DrawRectangle(0, 0, width, height);
111  int imgx = width/2+deltaX-zwidth/2;
112  int imgy = height/2+deltaY-zheight/2;
113  if (image!=NULL) {
114  wxImage drawimg = image->Copy();
115  drawimg.Rescale(zwidth,zheight,wxIMAGE_QUALITY_NORMAL);
116  dc.DrawBitmap(wxBitmap(drawimg),imgx,imgy);
117  dc.SetPen(*wxBLACK);
118  int bx = width/2+deltaX-zwidth/2;
119  int by = height/2+deltaY-zheight/2;
120  dc.DrawLine(bx,by,bx+zwidth,by);
121  dc.DrawLine(bx,by,bx,by+zheight);
122  dc.DrawLine(bx,by+zheight,bx+zwidth,by+zheight);
123  dc.DrawLine(bx+zwidth,by,bx+zwidth,by+zheight);
124  }
125  wxBitmap scalepanelbmp(*scalepanel->getImage());
126  scalepanelbmp = scalepanelbmp.Rescale(0,0,10000000,1000000,zwidth,zheight);
127  dc.DrawBitmap(scalepanelbmp,imgx,imgy);
128  wxPoint img_pos(imgx,imgy);
129  scalepanel->paintTo(dc,zoom,img_pos);
130 
131  wxImage statusimg = wxImage(width,30,true); //temporäres bild für Statusleiste
132  statusimg.InitAlpha();
133  for (int i=0;i<30*width;i++) {
134  statusimg.SetAlpha(i%width,i/width,150);
135  }
136  wxBitmap statusbmp(statusimg);
137  dc.DrawBitmap(statusbmp,0,height-30);
138  dc.SetTextForeground(wxColour(255,255,255));
139  if (current_mx>=imgx && current_mx<imgx+zwidth && current_my>=imgy && current_my<imgy+zheight) {
140  int x = int((current_mx-imgx)/zoom);
141  int y = int((current_my-imgy)/zoom);
142  float value = 0;
143  if (value_img!=NULL) {
144  value = value_img[y*imgwidth+x];
145  }
146  wxString val_str = floattowxstr(value)+wxT("°C");
147  if (abs(value+300)<.0001) {
148  val_str = wxT(" - ");
149  }
150  value = int(value*100.)/100.;
151  dc.DrawText(wxT("X:")+floattowxstr(x)+wxT(" Y: ")+floattowxstr(y)+wxT(" Wert: ")+val_str,20,height-25);
152  }
