112  void parseLine(string line,vector<string> &out,vector<string>* timestamps,vector<string>* names,vector<int>* valid_cols) {
113  string workstr = line.substr(0,line.length());
114  out.clear();
115  size_t sep_pos = workstr.find(opts.separator);
116  size_t count = 0;
117  while (sep_pos!=workstr.npos) {
118  sep_pos = workstr.find(opts.separator);
119  string substr = workstr.substr(0,sep_pos);
120  workstr.erase(0,sep_pos+1);
121  if (substr=="") continue;
122  if (count>=opts.start_col) {
123  if (valid_cols==NULL || contains(*valid_cols,count-opts.start_col)) {
124  out.resize(out.size()+1,substr);
125  }
126  }
127  if (count==opts.timecol && timestamps!=NULL) {
128  timestamps->resize(timestamps->size()+1,substr);
129  }
130  if (count==opts.namecol && names!=NULL) {
131  names->resize(names->size()+1,substr);
132  }
133  count++;
134  }
135  }
