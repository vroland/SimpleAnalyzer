57 void GUIColorScalePanel::refresh(int img_width,int img_height) {
58  if (mode!=SCM_NONE) {
59  wxRect display_area;
60  getDisplayArea(&display_area,1.0);
61  wxBitmap color_bmp = wxBitmap(img_width,img_height);
62  wxBitmap alpha_bmp = wxBitmap(img_width,img_height);
63  wxMemoryDC col_dc(color_bmp);
64  wxMemoryDC alpha_dc(alpha_bmp);
65  col_dc.SetBrush(text_color);
66  col_dc.SetPen(text_color);
67  alpha_dc.SetBrush(*wxBLACK_BRUSH);
68  alpha_dc.SetPen(*wxBLACK_PEN);
69  col_dc.DrawRectangle(0,0,img_width,img_height);
70  alpha_dc.DrawRectangle(0,0,img_width,img_height);
71  wxImage temp_color_img = wxImage(display_area.width,display_area.height);
72  for (int i=0;i<display_area.width;i++) {
73  for (int j=0;j<display_area.height;j++) {
74  float* color = hsvToRgb((1.-(mode==SCM_HORIZONTAL?i/(float)display_area.width:j/(float)display_area.height))*.66666,1,1);
75  temp_color_img.SetRGB(i,j,color[0]*255,color[1]*255,color[2]*255);
76  delete[] color;
77  }
78  }
79 
80  col_dc.SetBrush(*wxWHITE_BRUSH);
81  col_dc.SetPen(*wxBLACK_PEN);
82  alpha_dc.SetBrush(*wxWHITE_BRUSH);
83  alpha_dc.SetPen(*wxWHITE_PEN);
84  alpha_dc.DrawRectangle(display_area);
85  col_dc.DrawBitmap(wxBitmap(temp_color_img),display_area.x,display_area.y);
86 
87  //wxFont* font = wxFont::New(font_size*zoom, wxFONTFAMILY_DEFAULT, wxFONTFLAG_NOT_ANTIALIASED);
88  wxFont font(int(font_size),wxFONTFAMILY_DEFAULT,wxFONTSTYLE_NORMAL,wxFONTSTYLE_NORMAL);
89  col_dc.SetFont(font);
90  col_dc.SetTextForeground(text_color);
91  col_dc.SetPen(wxPen(text_color));
92  alpha_dc.SetFont(font);
93  alpha_dc.SetTextForeground(wxColour(255,255,255));
94  alpha_dc.SetPen(*wxWHITE_PEN);
95  int delta_t_vis = (visualization_info.max_visualisation_temp-visualization_info.min_visualisation_temp);
96  for (int i=0;i<=(delta_t_vis-delta_t_vis%step_width)/step_width;i++) {
97  wxString text = floattowxstr(int(visualization_info.min_visualisation_temp+i*step_width));
98  wxSize t_size = col_dc.GetTextExtent(text);
99  switch (mode) {
100  case SCM_VERTICAL: {
101  float display_step = (float)display_area.height/(delta_t_vis)*step_width;
102  wxPoint textpoint(display_area.x+display_area.width,display_area.y+i*display_step);
103  col_dc.DrawText(text,textpoint.x+8,textpoint.y-t_size.y/2.);
104  col_dc.DrawLine(textpoint.x,textpoint.y,textpoint.x+5,textpoint.y);
105  alpha_dc.DrawText(text,textpoint.x+8,textpoint.y-t_size.y/2.);
106  alpha_dc.DrawLine(textpoint.x,textpoint.y,textpoint.x+5,textpoint.y);
107  break;
108  }
109  case SCM_HORIZONTAL: {
110  float display_step = (float)display_area.width/(delta_t_vis)*step_width;
111  wxPoint textpoint(display_area.x+i*display_step,display_area.y+display_area.height);
112  col_dc.DrawText(text,textpoint.x-t_size.x/2.,textpoint.y+8);
113  col_dc.DrawLine(textpoint.x,textpoint.y,textpoint.x,textpoint.y+5);
114  alpha_dc.DrawText(text,textpoint.x-t_size.x/2.,textpoint.y+8);
115  alpha_dc.DrawLine(textpoint.x,textpoint.y,textpoint.x,textpoint.y+5);
116  break;
117  }
118  default:
119  break;
120  }
121  }
122  col_dc.SelectObject(wxNullBitmap);
123  alpha_dc.SelectObject(wxNullBitmap);
124  delete image;
125  image = new wxImage(color_bmp.GetWidth(),color_bmp.GetHeight());
126  image->InitAlpha();
127  wxImage alpha_img = alpha_bmp.ConvertToImage();
128  wxImage color_img = color_bmp.ConvertToImage();
129  for (int x=0;x<image->GetWidth();x++) {
130  for (int y=0;y<image->GetHeight();y++) {
131  image->SetRGB(x,y,color_img.GetRed(x,y),color_img.GetGreen(x,y),color_img.GetBlue(x,y));
132  image->SetAlpha(x,y,alpha_img.GetRed(x,y));
133  }
134  }
135  } else {
136  delete image;
137  image = new wxImage(img_width,img_height);
138  image->InitAlpha();
139  for (int i=0;i<img_width;i++) {
140  for (int j=0;j<img_height;j++) {
141  float* color = hsvToRgb((1.-(mode==SCM_HORIZONTAL?i/(float)img_width:j/(float)img_height))*.66666,1,1);
142  image->SetRGB(i,j,0,0,0);
143  image->SetAlpha(i,j,0);
144  delete[] color;
145  }
146  }
147  }

170 void GUIColorScalePanel::fitBounds(wxPoint& img_dim,bool to_scale) {
171  bool changed = 0;
172  if (x<0) {
173  x = 0;
174  changed = 1;
175  }
176  if (y<0) {
177  y = 0;
178  changed = 1;
179  }
180  if (width<MIN_WIDTH) {
181  width = MIN_WIDTH;
182  changed = 1;
183  }
184  if (height<MIN_HEIGHT) {
185  height = MIN_HEIGHT;
186  changed = 1;
187  }
188  if (width>img_dim.x) {
189  width = img_dim.x;
190  changed = 1;
191  }
192  if (height>img_dim.y) {
193  height = img_dim.y;
194  changed = 1;
195  }
196  if (to_scale && !(img_dim.x-x<MIN_WIDTH || img_dim.y-y<MIN_HEIGHT)) { // image dimensions must not me lower than MIN_WIDTH/HEIGHT
197  if (x+width>img_dim.x) {
198  width = img_dim.x-x;
199  changed = 1;
200  }
201  if (y+height>img_dim.y) {
202  height = img_dim.y-y;
203  changed = 1;
204  }
205  } else {
206  if (x+width>img_dim.x) {
207  x = img_dim.x-width;
208  changed = 1;
209  }
210  if (y+height>img_dim.y) {
211  y = img_dim.y-height;
212  changed = 1;
213  }
214  }
215  if (changed)
216  refresh(img_dim.x,img_dim.y);
217 }
218 void GUIColorScalePanel::handleMouse(wxMouseEvent& event,wxPoint& img_coords,wxPoint& img_dim,float zoom) {
219  if (mode!=SCM_NONE) {
220  wxRect display_area;
221  getDisplayArea(&display_area,zoom);
222  display_area.x+=img_coords.x;
223  display_area.y+=img_coords.y;
224  if (prev_mouse_down!=event.m_leftDown && !prev_mouse_down) {
225  wxPoint sc_topleft(display_area.GetBottomRight()-wxPoint(10,10));
226  wxRect scalearea = wxRect(sc_topleft,display_area.GetBottomRight());
227  if (scalearea.Contains(event.GetPosition())) {
228  scaling = true;
229  } else {
230  scaling = false;
231  }
232  } else {
233  if (event.m_leftDown) {
234  if (scaling) {
235  width+=(event.m_x-current_mx)/zoom;
236  height+=(event.m_y-current_my)/zoom;
237  fitBounds(img_dim,true);
238  } else {
239  x+=(event.m_x-current_mx)/zoom;
240  y+=(event.m_y-current_my)/zoom;
241  fitBounds(img_dim,false);
242  }
243  }
244  }
245  transforming = event.m_leftDown;
246  }
247  if (prev_mouse_down!=event.m_leftDown) {
248  refresh(img_dim.x,img_dim.y);
249  }
250  prev_mouse_down = event.m_leftDown;
251  current_mx = event.m_x;
252  current_my = event.m_y;
253 }
