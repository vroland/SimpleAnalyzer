85 void GUIColorScalePanel::refresh(int img_width, int img_height) {
86 
87  //Soll die Skala dargestellt werden?
88  if (mode != SCM_NONE) {
89  //Fläche des Farbbandes
90  wxRect display_area;
91  getDisplayArea(&display_area, 1.0);
92 
93  //Temporäre Zwischenbilder für Farbe und Transparenz
94  wxBitmap color_bmp = wxBitmap(img_width, img_height);
95  wxBitmap alpha_bmp = wxBitmap(img_width, img_height);
96 
97  //Erstellen von DeviceContexts zum Zeichnen auf die bilder
98  wxMemoryDC col_dc(color_bmp);
99  wxMemoryDC alpha_dc(alpha_bmp);
100 
101  //Schwarzfärben der Bilder (entspricht im Transparentbild volltransparent)
102  col_dc.SetBrush(text_color);
103  col_dc.SetPen(text_color);
104  alpha_dc.SetBrush(*wxBLACK_BRUSH);
105  alpha_dc.SetPen(*wxBLACK_PEN);
106  col_dc.DrawRectangle(0, 0, img_width, img_height);
107  alpha_dc.DrawRectangle(0, 0, img_width, img_height);
108 
109  //Temporäres Bild für das Farbband
110  wxImage temp_color_img = wxImage(display_area.width,
111  display_area.height);
112 
113  //Für jedes Pixel des Farbbands
114  for (int i = 0; i < display_area.width; i++) {
115  for (int j = 0; j < display_area.height; j++) {
116  //Zwischenvariable zur Ermittlung des Farbwerts
117  float inverse_hue = 0;
118 
119  //Ist die Skala Horizontal?
120  if (mode == SCM_HORIZONTAL) {
121  //Hhorizontaler Farbverlauf
122  inverse_hue = i / (float) display_area.width;
123  } else {
124  //vertikaler Farbverlauf
125  inverse_hue = j / (float) display_area.height;
126  }
127 
128  //Berechnen der Farbe
129  float* color = hsvToRgb((1. - inverse_hue) * .66666, 1, 1);
130  //Einfärben des Pixels
131  temp_color_img.SetRGB(i, j, color[0] * 255, color[1] * 255,
132  color[2] * 255);
133  delete[] color;
134  }
135  }
136 
137  col_dc.SetBrush(*wxWHITE_BRUSH);
138  col_dc.SetPen(*wxBLACK_PEN);
139  //Weiße Farbe auf dem Bild für die Transparenz heißt undurchsichtig
140  alpha_dc.SetBrush(*wxWHITE_BRUSH);
141  alpha_dc.SetPen(*wxWHITE_PEN);
142 
143  //Undurchsichtigmachen des Bereichs für das Farbband
144  alpha_dc.DrawRectangle(display_area);
145  //Zeichnen des Farbbands auf das Farbbild
146  col_dc.DrawBitmap(wxBitmap(temp_color_img), display_area.x,
147  display_area.y);
148 
149  //Erstellen einer Schrift mit der gewünschten Schriftgröße
150  wxFont font(int(font_size), wxFONTFAMILY_DEFAULT, wxFONTSTYLE_NORMAL,
151  wxFONTSTYLE_NORMAL);
152 
153  //Übertragen der Einstellungen für die Schrift in die DCs
154  col_dc.SetFont(font);
155  col_dc.SetTextForeground(text_color);
156  col_dc.SetPen(wxPen(text_color));
157  alpha_dc.SetFont(font);
158  alpha_dc.SetTextForeground(wxColour(255, 255, 255));
159  alpha_dc.SetPen(*wxWHITE_PEN);
160 
161  //Die Visualisierungseinstellungen
162  Visualization_info* vis_info = wxGetApp().getVisualizationInfo();
163 
164  //Spannweite des darzustellenden Temperaturbereichs
165  int delta_t_vis = (vis_info->max_visualisation_temp
166  - vis_info->min_visualisation_temp);
167 
168  //Beschriftung der Skala:
169 
170  //So viele Schritte, wie mit der angegebenen Schrittweite auf die Skala passen...
171  for (int i = 0; i <= (delta_t_vis - delta_t_vis % step_width) / step_width; i++) {
172 
173  //Beschriftungstext an dieser Stelle
174  wxString text = floattowxstr(
175  int(vis_info->min_visualisation_temp + i * step_width));
176  //Größe des Schriftzuges
177  wxSize t_size = col_dc.GetTextExtent(text);
178 
179  //Darstellungsmodus?
180  switch (mode) {
181  case SCM_VERTICAL: {
182  //Pixel pro Schritt
183  float display_step = (float) display_area.height / (delta_t_vis)
184  * step_width;
185  //Position des Beschriftungstexts
186  wxPoint textpoint(display_area.x + display_area.width,
187  display_area.y + i * display_step);
188 
189  //Zeichnen des Texts und Markierungsstrichs auf Farb- und Transparenzbild
190  col_dc.DrawText(text, textpoint.x + 8,
191  textpoint.y - t_size.y / 2.);
192  col_dc.DrawLine(textpoint.x, textpoint.y, textpoint.x + 5,
193  textpoint.y);
194  alpha_dc.DrawText(text, textpoint.x + 8,
195  textpoint.y - t_size.y / 2.);
196  alpha_dc.DrawLine(textpoint.x, textpoint.y, textpoint.x + 5,
197  textpoint.y);
198  break;
199  }
200  case SCM_HORIZONTAL: {
201  //Pixel pro Schritt
202  float display_step = (float) display_area.width / (delta_t_vis)
203  * step_width;
204  //Position des Beschriftungstexts
205  wxPoint textpoint(display_area.x + i * display_step,
206  display_area.y + display_area.height);
207 
208  //Zeichnen des Texts und Markierungsstrichs auf Farb- und Transparenzbild
209  col_dc.DrawText(text, textpoint.x - t_size.x / 2.,
210  textpoint.y + 8);
211  col_dc.DrawLine(textpoint.x, textpoint.y, textpoint.x,
212  textpoint.y + 5);
213  alpha_dc.DrawText(text, textpoint.x - t_size.x / 2.,
214  textpoint.y + 8);
215  alpha_dc.DrawLine(textpoint.x, textpoint.y, textpoint.x,
216  textpoint.y + 5);
217  break;
218  }
219  default:
220  break;
221  }
222  }
223 
224  //lösen der Zwischenbilder von den DCs
225  col_dc.SelectObject(wxNullBitmap);
226  alpha_dc.SelectObject(wxNullBitmap);
227 
228  //zurücksetzen des Ausgabebilds
229  delete image;
230  image = new wxImage(color_bmp.GetWidth(), color_bmp.GetHeight());
231  image->InitAlpha();
232 
233  //Zwischenbilder als wxImage, um einzelne Pixels auslesbar zu machen
234  wxImage alpha_img = alpha_bmp.ConvertToImage();
235  wxImage color_img = color_bmp.ConvertToImage();
236 
237  //Übertragen von Farbe und Transparenz in das Ausgabebild
238  for (int x = 0; x < image->GetWidth(); x++) {
239  for (int y = 0; y < image->GetHeight(); y++) {
240  image->SetRGB(x, y, color_img.GetRed(x, y),
241  color_img.GetGreen(x, y), color_img.GetBlue(x, y));
242  image->SetAlpha(x, y, alpha_img.GetRed(x, y));
243  }
244  }
245  } else {
246  //zurücksetzen des Ausgabebilds
247  delete image;
248  image = new wxImage(img_width, img_height);
249  image->InitAlpha();
250 
251  //Ausgabebild volltransparent machen
252  for (int i = 0; i < img_width; i++) {
253  for (int j = 0; j < img_height; j++) {
254  image->SetRGB(i, j, 0, 0, 0);
255  image->SetAlpha(i, j, 0);
256  }
257  }
258  }
259 }

296 void GUIColorScalePanel::fitBounds(wxPoint& img_dim, bool to_scale) {
297 
298  //hat eine Veränderung stattgefunden?
299  bool changed = 0;
300 
301  //Position gültig?
302  if (x < 0) {
303  x = 0;
304  changed = 1;
305  }
306  if (y < 0) {
307  y = 0;
308  changed = 1;
309  }
310 
311  //Größe gültig?
312  if (width < MIN_WIDTH) {
313  width = MIN_WIDTH;
314  changed = 1;
315  }
316  if (height < MIN_HEIGHT) {
317  height = MIN_HEIGHT;
318  changed = 1;
319  }
320  if (width > img_dim.x) {
321  width = img_dim.x;
322  changed = 1;
323  }
324  if (height > img_dim.y) {
325  height = img_dim.y;
326  changed = 1;
327  }
328 
329  //Soll oder muss die Skalengröße verändert werden?
330  if (to_scale
331  && !(img_dim.x - x < MIN_WIDTH || img_dim.y - y < MIN_HEIGHT)) {
332 
333  //Anpassen der Skalengröße an die Bildgröße
334  if (x + width > img_dim.x) {
335  width = img_dim.x - x;
336  changed = 1;
337  }
338  if (y + height > img_dim.y) {
339  height = img_dim.y - y;
340  changed = 1;
341  }
342  } else {
343  //Anpassen der Sklenposition an das Bild
344  if (x + width > img_dim.x) {
345  x = img_dim.x - width;
346  changed = 1;
347  }
348  if (y + height > img_dim.y) {
349  y = img_dim.y - height;
350  changed = 1;
351  }
352  }
353 
354  //hat eine Veränderung stattgefunden?
355  if (changed) {
356  //Skale neu berechnen
357  refresh(img_dim.x, img_dim.y);
358  }
359 }
360 void GUIColorScalePanel::handleMouse(wxMouseEvent& event, wxPoint& img_coords,
361  wxPoint& img_dim, float zoom) {
362 
363  //Soll die Skala dargestellt werden?
364  if (mode != SCM_NONE) {
365  //Fläche des Farbbands
366  wxRect display_area;
367  getDisplayArea(&display_area, zoom);
368 
369  //auf Fensterkoordinaten umrechnen
370  display_area.x += img_coords.x;
371  display_area.y += img_coords.y;
372 
373  //Ist die linke Maustaste gedrückt worden?
374  if (prev_mouse_down != event.m_leftDown && !prev_mouse_down) {
375  //Fläche zum Skalieren
376  wxPoint sc_topleft(display_area.GetBottomRight() - wxPoint(10, 10));
377  wxRect scalearea = wxRect(sc_topleft,
378  display_area.GetBottomRight());
379 
380  //wurde auf die Fläche zum Skalieren geklickt?
381  if (scalearea.Contains(event.GetPosition())) {
382  scaling = true;
383  } else {
384  scaling = false;
385  }
386  } else {
387  //Ist die linke Maustaste gedrückt?
388  if (event.m_leftDown) {
389  //wird gerade skaliert?
390  if (scaling) {
391  //Skala skalieren
392  width += (event.m_x - current_mx) / zoom;
393  height += (event.m_y - current_my) / zoom;
394  fitBounds(img_dim, true);
395  } else {
396  //Skala verschieben
397  x += (event.m_x - current_mx) / zoom;
398  y += (event.m_y - current_my) / zoom;
399  fitBounds(img_dim, false);
400  }
401  }
402  }
403 
404  //findet eine Transformation statt?
405  transforming = event.m_leftDown;
406  }
407 
408  //wurde geklickt oder die linke Maustaste losgelassen?
409  if (prev_mouse_down != event.m_leftDown) {
410  //Skalenbild aktualisieren
411  refresh(img_dim.x, img_dim.y);
412  }
413 
414  //vorherigen Maussatus speichern
415  prev_mouse_down = event.m_leftDown;
416  //vorherige Mausposition speichern
417  current_mx = event.m_x;
418  current_my = event.m_y;
419 }
