62  file << io->pointlist[p * 3] << " " << io->pointlist[p * 3 + 1]
63  << " " << io->pointlist[p * 3 + 2] << " " << endl;
64  }
65  }
66 
67  //Alle Flächen schreiben
68  file << "POLYGONS " << tetrahedra_count * 4 << " "
69  << tetrahedra_count * 4 * 4 << endl;
70  //Zähler, der auf die Punktindices addiert wird um Überschneidung zu verhindern
71  int material_point_counter = 0;
72 
73  //Schreiben der Flächen für alle Materialien
74  for (int m = 0; m < material_count; m++) {
75  tetgenio* io = data->getMaterials()->at(m).tetgenoutput;
76 
77  for (int t = 0; t < io->numberoftetrahedra; t++) {
78  for (int k = 0; k < 4; k++) {
79  file << 3 << " "
80  << material_point_counter
81  + io->tetrahedronlist[4 * t
82  + tetface_indices[k][0]] << " "
83  << material_point_counter
84  + io->tetrahedronlist[4 * t
85  + tetface_indices[k][1]] << " "
86  << material_point_counter
87  + io->tetrahedronlist[4 * t
88  + tetface_indices[k][2]] << endl;
89  }
90  }
91 
92  //Indexbereich für das nächste Material
93  material_point_counter += io->numberofpoints;
94  }
95 
96  //Schreiben der Temperaturwerte für alle Punkte
97  file << "POINT_DATA " << point_count << endl;
98  file << "SCALARS temperature float" << endl;
99  file << "LOOKUP_TABLE default" << endl;
100  for (int m = 0; m < material_count; m++) {
101  tetgenio* io = data->getMaterials()->at(m).tetgenoutput;
102  for (int p = 0; p < io->numberofpoints; p++) {
103  file << io->pointattributelist[p * io->numberofpointattributes]
104  << endl;
105  }
106  }

112  tetgenio* io = data->getMaterials()->at(m).tetgenoutput;
